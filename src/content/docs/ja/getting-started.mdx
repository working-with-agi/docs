## AI ターミナル

<Tabs>
  <TabItem label="React">
  ```tsx
  import { AgiTerminal } from "@work-with-ai/react";

  export default function Terminal() {
    return (
      <AgiTerminal
        endpoint="https://your-server:8600"
        apiKey={process.env.NEXT_PUBLIC_AGI_KEY}
        theme="dark"
        onConnect={() => console.log("Connected")}
        onDisconnect={() => console.log("Disconnected")}
        onOutput={(data) => console.log("Output:", data)}
      />
    );
  }
  ```
  </TabItem>
  <TabItem label="Vue">
  ```vue
  <script setup>
  import { AgiTerminal } from "@work-with-ai/vue";
  </script>

  <template>
    <AgiTerminal
      endpoint="https://your-server:8600"
      :apiKey="agiKey"
      theme="dark"
      @connect="() => console.log('Connected')"
      @disconnect="() => console.log('Disconnected')"
      @output="(data) => console.log('Output:', data)"
    />
  </template>
  ```
  </TabItem>
  <TabItem label="Angular">
  ```typescript
  import { Component } from "@angular/core";
  import { AgiTerminalComponent } from "@work-with-ai/angular";

  @Component({
    imports: [AgiTerminalComponent],
    template: `
      <agi-terminal
        [endpoint]="endpoint"
        [apiKey]="apiKey"
        theme="dark"
        (connected)="onConnected()"
        (disconnected)="onDisconnected()"
        (output)="onOutput($event)"
      />
    `,
  })
  export class TerminalPage {
    endpoint = "https://your-server:8600";
    apiKey = environment.agiKey;
    onConnected() { console.log("Connected"); }
    onDisconnected() { console.log("Disconnected"); }
    onOutput(data: any) { console.log("Output:", data); }
  }
  ```
  </TabItem>
  <TabItem label="Vanilla JS">
  ```typescript
  import { WorkAGI } from "@work-with-ai/sdk";

  const terminal = WorkAGI.terminal({
    container: "#terminal",
    endpoint: "https://your-server:8600",
    apiKey: "your-api-key",
    theme: "dark",
    onConnect: () => console.log("Connected"),
    onDisconnect: () => console.log("Disconnected"),
    onOutput: (data) => console.log("Output:", data),
  });

  // プログラムからの操作
  await terminal.execute("ls -la");
  terminal.dispose();
  ```
  </TabItem>

## KnowledgeHub

```typescript
import { KnowledgeClient } from "@work-with-ai/sdk";

const kb = new KnowledgeClient("https://your-hub:8610", "your-api-key");
const results = await kb.search("error handling for API timeouts");
// → [{ source, section, score, snippet }]
```

## セルフホスト

バックエンドインフラを自分のサーバーで実行する場合:

```bash
# スタックをクローン
git clone https://github.com/working-with-agi/self-hosted.git
cd self-hosted

# 全サービス起動（ターミナルサーバー + KnowledgeHub）
docker compose up -d

# ナレッジソースを取り込み
docker compose exec knowledgehub ingest /path/to/your/content
```
