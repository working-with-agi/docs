## KnowledgeHub

<Tabs>
  <TabItem label="React">
  ```tsx
  import { useKnowledgeHub } from "@working-with-agi/react";

  export default function Search() {
    const { search, results, loading } = useKnowledgeHub({
      apiKey: process.env.NEXT_PUBLIC_AGI_KEY,
      index: "my-project",
    });

    return (
      <div>
        <input onChange={(e) => search(e.target.value)} placeholder="ナレッジを検索..." />
        {results.map((hit) => (
          <div key={hit.source + hit.section}>
            <code>{hit.source}:{hit.section}</code> — {hit.snippet}
          </div>
        ))}
      </div>
    );
  }
  ```
  </TabItem>
  <TabItem label="Vue">
  ```vue
  <script setup>
  import { useKnowledgeHub } from "@working-with-agi/vue";

  const { search, results, loading } = useKnowledgeHub({
    apiKey: agiKey,
    index: "my-project",
  });
  </script>

  <template>
    <input @input="search($event.target.value)" placeholder="ナレッジを検索..." />
    <div v-for="hit in results" :key="hit.source + hit.section">
      <code>{{ hit.source }}:{{ hit.section }}</code> — {{ hit.snippet }}
    </div>
  </template>
  ```
  </TabItem>
  <TabItem label="Angular">
  ```typescript
  import { Component } from "@angular/core";
  import { KnowledgeHubService } from "@working-with-agi/angular";

  @Component({
    template: `
      <input (input)="onSearch($event)" placeholder="ナレッジを検索..." />
      <div *ngFor="let hit of results">
        <code>{{ hit.source }}:{{ hit.section }}</code> — {{ hit.snippet }}
      </div>
    `,
  })
  export class SearchPage {
    results: any[] = [];
    constructor(private knowledgeHub: KnowledgeHubService) {}
    async onSearch(event: Event) {
      const query = (event.target as HTMLInputElement).value;
      this.results = await this.knowledgeHub.search(query);
    }
  }
  ```
  </TabItem>
  <TabItem label="Vanilla JS">
  ```typescript
  import { WorkAGI } from "@working-with-agi/sdk";

  const hub = WorkAGI.knowledgeHub({
    apiKey: "your-api-key",
    index: "my-project",
  });

  const results = await hub.search("API タイムアウトのエラーハンドリング");
  // → [{ source, section, score, snippet }]
  ```
  </TabItem>

## セルフホスト

バックエンドインフラを自分のサーバーで実行する場合:

```bash
# スタックをクローン
git clone https://github.com/working-with-agi/self-hosted.git
cd self-hosted

# 全サービス起動（ターミナルサーバー + KnowledgeHub）
docker compose up -d

# ナレッジソースを取り込み
docker compose exec knowledgehub ingest /path/to/your/content
```
