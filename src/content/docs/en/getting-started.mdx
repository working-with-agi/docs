## AI Terminal

<Tabs>
  <TabItem label="React">
  ```tsx
  import { AgiTerminal } from "@working-with-agi/react";

  export default function Terminal() {
    return (
      <AgiTerminal
        apiKey={process.env.NEXT_PUBLIC_AGI_KEY}
        theme="dark"
        ai
        recording
        onCommand={(cmd) => console.log("Command:", cmd)}
        onThreat={(alert) => notifyAdmin(alert)}
      />
    );
  }
  ```
  </TabItem>
  <TabItem label="Vue">
  ```vue
  <script setup>
  import { AgiTerminal } from "@working-with-agi/vue";
  </script>

  <template>
    <AgiTerminal
      :api-key="agiKey"
      theme="dark"
      ai
      recording
      @command="(cmd) => console.log('Command:', cmd)"
      @threat="(alert) => notifyAdmin(alert)"
    />
  </template>
  ```
  </TabItem>
  <TabItem label="Angular">
  ```typescript
  import { Component } from "@angular/core";
  import { AgiTerminalComponent } from "@working-with-agi/angular";

  @Component({
    imports: [AgiTerminalComponent],
    template: `
      <agi-terminal
        [apiKey]="apiKey"
        theme="dark"
        [ai]="true"
        [recording]="true"
        (command)="onCommand($event)"
        (threat)="onThreat($event)"
      />
    `,
  })
  export class TerminalPage {
    apiKey = environment.agiKey;
    onCommand(cmd: string) { console.log("Command:", cmd); }
    onThreat(alert: any) { notifyAdmin(alert); }
  }
  ```
  </TabItem>
  <TabItem label="Vanilla JS">
  ```typescript
  import { WorkAGI } from "@working-with-agi/sdk";

  const terminal = WorkAGI.terminal("#terminal", {
    apiKey: "your-api-key",
    theme: "dark",
    ai: true,
    recording: true,
    onCommand: (cmd) => console.log("Command:", cmd),
    onThreat: (alert) => notifyAdmin(alert),
  });

  // Programmatic control
  await terminal.execute("ls -la");
  terminal.dispose();
  ```
  </TabItem>

## RAG Code Search

<Tabs>
  <TabItem label="React">
  ```tsx
  import { useCodeSearch } from "@working-with-agi/react";

  export default function Search() {
    const { search, results, loading } = useCodeSearch({
      apiKey: process.env.NEXT_PUBLIC_AGI_KEY,
      index: "my-project",
    });

    return (
      <div>
        <input onChange={(e) => search(e.target.value)} placeholder="Search code..." />
        {results.map((hit) => (
          <div key={hit.file + hit.line}>
            <code>{hit.file}:{hit.line}</code> — {hit.snippet}
          </div>
        ))}
      </div>
    );
  }
  ```
  </TabItem>
  <TabItem label="Vue">
  ```vue
  <script setup>
  import { useCodeSearch } from "@working-with-agi/vue";

  const { search, results, loading } = useCodeSearch({
    apiKey: agiKey,
    index: "my-project",
  });
  </script>

  <template>
    <input @input="search($event.target.value)" placeholder="Search code..." />
    <div v-for="hit in results" :key="hit.file + hit.line">
      <code>{{ hit.file }}:{{ hit.line }}</code> — {{ hit.snippet }}
    </div>
  </template>
  ```
  </TabItem>
  <TabItem label="Angular">
  ```typescript
  import { Component } from "@angular/core";
  import { CodeSearchService } from "@working-with-agi/angular";

  @Component({
    template: `
      <input (input)="onSearch($event)" placeholder="Search code..." />
      <div *ngFor="let hit of results">
        <code>{{ hit.file }}:{{ hit.line }}</code> — {{ hit.snippet }}
      </div>
    `,
  })
  export class SearchPage {
    results: any[] = [];
    constructor(private codeSearch: CodeSearchService) {}
    async onSearch(event: Event) {
      const query = (event.target as HTMLInputElement).value;
      this.results = await this.codeSearch.search(query);
    }
  }
  ```
  </TabItem>
  <TabItem label="Vanilla JS">
  ```typescript
  import { WorkAGI } from "@working-with-agi/sdk";

  const search = WorkAGI.codeSearch({
    apiKey: "your-api-key",
    index: "my-project",
  });

  const results = await search.query("error handling for API timeouts");
  // → [{ file, line, score, snippet }]
  ```
  </TabItem>

## Self-Hosted Setup

To run the backend infrastructure on your own servers:

```bash
# Clone the stack
git clone https://github.com/working-with-agi/self-hosted.git
cd self-hosted

# Start everything (terminal server + RAG pipeline + Qdrant)
docker compose up -d

# Index your codebase
docker compose exec indexer grepai index /path/to/your/code
```
